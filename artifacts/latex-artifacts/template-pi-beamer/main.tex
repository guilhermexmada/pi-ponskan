\begin{frame}{Estado da Arte}
\centering

\vspace{0.2cm}
{\color{green}\large\textbf{CITRUSVIS - Um sistema de visão computacional para a identificação do fungo Guignardia citricarpa, causador da mancha preta em citros}}\\[0.2cm]
\normalsize {Pazoti, 2005} \\[0.2cm]

\begin{block}{Rede Neural Artificial (RNA)}
  Aprendizado supervisionado com 2 camadas escondidas de 20 e 15 neurônios.
\end{block}

\begin{block}{Resultados}
  98\% de acurácia em amostras pré-selecionadas e 96,6\% em testes gerais, com 66 segundos de tempo de processamento médio.
\end{block}

\begin{block}{Desafios}
  Limitações e ruídos na aquisição de imagens.
\end{block}

\section{Objetivo}
\begin{frame}{Metodologia confusa ou incompleta}
\justifying
Mistura de etapas, ferramentas e resultados.  
Falta de clareza sobre o fluxo de desenvolvimento, separação entre metodologia de pesquisa e de implementação, e presença de trechos do modelo sem conteúdo autoral.
\end{frame}

\section{Apresentação Prática}
\begin{frame}{Citações mal posicionadas ou incompletas}
\justifying
Pedidos de “citação ao final do parágrafo” ignorados.  
Falta de identificação clara da fonte de dados e lacunas entre citações e lista de referências.
\end{frame}
